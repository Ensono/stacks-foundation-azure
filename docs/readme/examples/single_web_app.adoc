=== Single Project Web Application

[source,hcl]
----
module "foundation" {
  source       = "./stacks-foundation-azure"
  company_name = "Contoso"
  location     = "eastus"
  project      = ["webapp"]
  stage_name   = "foundation"
  environment  = "prod"
}

resource "azurerm_resource_group" "webapp" {
  name     = module.foundation.names["webapp"].resource_group.name
  location = var.location
}

resource "azurerm_app_service_plan" "webapp" {
  name                = module.foundation.names["webapp"].app_service_plan.name
  resource_group_name = azurerm_resource_group.webapp.name
  location            = azurerm_resource_group.webapp.location
  sku {
    tier = "Standard"
    size = "S1"
  }
}

resource "azurerm_app_service" "webapp" {
  name                = module.foundation.names["webapp"].app_service.name
  resource_group_name = azurerm_resource_group.webapp.name
  location            = azurerm_resource_group.webapp.location
  app_service_plan_id = azurerm_app_service_plan.webapp.id
}
----
